function bingo(ticket, win){
	let count = 0;
	ticket.forEach(array => {
		array[0].split('').forEach(letter => {
			if (letter.charCodeAt() == array[1]) {
				count = count + 1;
			}
		})
	});

	if (count >= win) {
		return 'Winner!'
	} else return 'Loser!'
}
